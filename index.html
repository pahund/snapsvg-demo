Snap.svg Demo 1

Make the circle move, change color and size: 

    var snap = Snap('#demo');
    var bigCircle = snap.circle(10, 100, 10);
    bigCircle.attr({
        fill: '#000000',
        stroke: '#000000',
        strokeWidth: 5
    });
    bigCircle.animate({
        cx: window.innerWidth / 2,
        r: 100,
        fill: '#ffa500'
    }, 1000, null, function () {
        bigCircle.animate({
            cx: window.innerWidth - 10,
            r: 10,
            fill: '#000000'
        }, 1000);
    });
